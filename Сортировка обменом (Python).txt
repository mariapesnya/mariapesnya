def bubble_sort(arr):
    n = len(arr)
    # Проходим по массиву n-1 раз
    for i in range(n):
        # В каждом проходе сравниваем соседние элементы
        for j in range(0, n - i - 1):
            # Если текущий элемент больше следующего, то меняем их местами
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
